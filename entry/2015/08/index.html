#include <memory>
#include <thread>

class nyan {
  std::unique_ptr<std::thread> th_;

public:
  void run() {
    th_ = std::make_unique<std::thread>(&nyan::worker, this);
  }

  ~nyan() {
    th_->join();
  }

private:
  void worker() {
    std::cout << "Nyan!!" << std::endl;
  }
};

auto main() -> int {
  nyan n;
  n.run();
}
